int change_filename(const char* old, const char* new)
{
	FILE *file;
	if(file = fopen(old,"r"))
	{
		printf("%s exists.\n",old);
        fclose(file);
    }
    else
    {
        printf("%s doesn't exist.\n",old);
		return 1;
    }
	return rename(old, new);
}

int list_files()
{
	struct dirent* de;
	
	DIR* dr = opendir(".");
	
	if(dr == NULL)
	{
		printf("cannot open current directory.");
		return 1;
	}
	
	while((de = readdir(dr)) != NULL)
	{
		printf("%s\n", de->d_name);
	}
	
	
	closedir(dr);   
	return 0;
}
